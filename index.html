21    const workouts = offlineWorkouts ? JSON.parse(offlineWorkouts) : [];
22    workouts.push({ ...workout, synced: false });
23    await AsyncStorage.setItem('offlineWorkouts', JSON.stringify(workouts));
24

27    }
28  }

30  async syncPendingData() {
31    const offlineWorkouts = await AsyncStorage.getItem('offlineWorkouts');
32    if (!offlineWorkouts) return;
33    
34    const workouts = JSON.parse(offlineWorkouts);
35    const unsynced = workouts.filter(w => !w.synced);
36    
37    for (const workout of unsynced) {
38      try {
39        await api.post('/workouts', workout);
40        workout.synced = true;
41      } catch (error) {
42        console.error('Sync failed:', error);
43      }
44    }
45    
46    await AsyncStorage.setItem('offlineWorkouts', JSON.stringify(workouts));
47  }
